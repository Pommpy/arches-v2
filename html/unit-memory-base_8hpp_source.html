147 void add_unit(Arches::paddr_t paddr, Arches::Units::UnitMemoryBase* unit, uint port_id, uint num_ports)
148 {
149 if(unit == nullptr)
150 {
151 num_ports = 0;
152 port_id = 0;
153 }
154
155 uint i = 0;
156 for(; i < ranges.size(); ++i)
157 if(paddr < ranges[i].first) break;
158
159 ranges.insert(ranges.begin() + i, {paddr, ~0u});
160
161 if(unit == nullptr) return;
162
163 MemoryMapping mapping = {unit, port_id, num_ports, 0};
164
165 uint j;
166 for(j = 0; j < mappings.size(); ++j)
167 if(mapping == mappings[j]) break;
168
169 if(j == mappings.size()) mappings.push_back(mapping);
170
171 ranges[i].second = j;
172
173 total_ports = 0;
174 for(j = 0; j < mappings.size(); ++j)
175 {
176 mappings[j].port_id = total_ports;
177 total_ports += mappings[j].num_ports;
178 }
179 }

181 uint get_mapping_index(Arches::paddr_t paddr)
182 {
183 uint start = 0;
184 uint end = ranges.size();
185 while((start + 1) != end)
186 {
187 uint middle = (start + end) / 2;
188 if(paddr >= ranges[middle].first) start = middle;
189 else end = middle;
190 }
191
192 return ranges[start].second;
193 }
