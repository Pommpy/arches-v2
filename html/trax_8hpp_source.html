35static paddr_t align_to(size_t alignment, paddr_t paddr)
36{
37 return (paddr + alignment - 1) & ~(alignment - 1);
38}

40template <typename RET>
41static RET* write_array(Units::UnitMainMemoryBase* main_memory, size_t alignment, RET* data, size_t size, paddr_t& heap_address)
42{
43 paddr_t array_address = align_to(alignment, heap_address);
44 heap_address = array_address + size * sizeof(RET);
45 main_memory->direct_write(data, size * sizeof(RET), array_address);
46 return reinterpret_cast<RET*>(array_address);
47}

49template <typename RET>
50static RET* write_vector(Units::UnitMainMemoryBase* main_memory, size_t alignment, std::vector<RET> v, paddr_t& heap_address)
51{
52 return write_array(main_memory, alignment, v.data(), v.size(), heap_address);
53}

55static GlobalData initilize_buffers(Units::UnitMainMemoryBase* main_memory, paddr_t& heap_address)
56{
57 GlobalData global_data;

62 heap_address = align_to(4 * 1024, heap_address);
63 global_data.framebuffer = reinterpret_cast<uint32_t*>(heap_address); heap_address += global_data.framebuffer_size * sizeof(uint32_t);

65 global_data.samples_per_pixel = 1;

72 Mesh mesh("benchmarks/trax/res/sponza2_triangulated.obj");
73 BVH blas;
74 std::vector<BuildObject> build_objects;
75 for(uint i = 0; i < mesh.size(); ++i)
76 build_objects.push_back(mesh.get_build_object(i));
77 blas.build(build_objects);
78 mesh.reorder(build_objects);
79
80 global_data.mesh.blas = write_vector(main_memory, CACHE_BLOCK_SIZE, blas.nodes, heap_address);
