143token_t tokenize(const char * input, bool &isInteger) {
144 const size_t numTokens = sizeof(tokenTranslatorTable) / sizeof(tokenTranslator);
145
146 const size_t lengthInput = strlen(input);
147 for(size_t i = 0; i < numTokens; ++i) {
148 // Special case for comments which are not separated by a space, aka //BLOB
149 const size_t lengthToCompare = (tokenTranslatorTable[i].token != comment_token ? lengthInput : strlen(tokenTranslatorTable[i].keyword));
150 if(strncmp(input, tokenTranslatorTable[i].keyword, lengthToCompare) == 0) {
151 isInteger = tokenTranslatorTable[i].isInteger;
152 return tokenTranslatorTable[i].token;
153 }
154 }
155
156 // Unknown token
157 printf("PANIC : Unknown token %s\n",input);
158 return unknown_token;
159}

162void read_config_file(FILE * fin)
163{
164 char c;
165 char input_string[256];
166 int input_int;
167 float input_float;
168
169 while ((c = fgetc(fin)) != EOF)
170 {
171 if ((c != EOL) &&
172 (c != CR) &&
173 (c != SPACE) &&
174 (c != TAB))
175 {
176 if (fscanf(fin, "%s", &input_string[1]) == 0)
177 continue;
178 input_string[0] = c;
179 }
180 else
181 {
182 if (fscanf(fin, "%s", &input_string[0]) == 0)
183 continue;
184 }
185 bool loadAsInt;
186 const token_t input_field = tokenize(&input_string[0], loadAsInt);
187
188 // badness
189 if (input_field == unknown_token)
190 {
191 printf("PANIC: bad token (%s) in cfg file\n", input_string);
192 continue;
193 }
194
195 // Ignore comment
196 if (input_field == comment_token)
197 {
198 while (((c = fgetc(fin)) != EOL) && (c != EOF))
199 {
200 /*comment, to be ignored */
201 }
202 continue;
203 }
204
205 // Attempt to read the data into correct type
206 if (loadAsInt)
207 {
208 if (fscanf(fin, "%d", &input_int) == 0)
209 {
210 printf("PANIC : token %s must have integer parameter value\n", input_string);
211 continue;
212 }
213 }
214 else
215 {
216 if (fscanf(fin, "%f", &input_float) == 0)
217 {
218 printf("PANIC : token %s must have float parameter value\n", input_string);
219 continue;
220 }
221 }
222
223 // Apply value correctly
224 switch (input_field)
225 {
226 case comment_token:
227 case unknown_token:
228 // already handled above
229 break;
230
231 case processor_clk_multiplier_token: PROCESSOR_CLK_MULTIPLIER = input_int; break;
232 case robsize_token: ROBSIZE = input_int; break;
233 case max_retire_token: MAX_RETIRE = input_int; break;
234 case max_fetch_token: MAX_FETCH = input_int; break;
235 case pipelinedepth_token: PIPELINEDEPTH = input_int; break;
236
237 case num_channels_token: NUM_CHANNELS = input_int; break;
238 case num_ranks_token: NUM_RANKS = input_int; break;
