38template <bool ascending = true, typename RET>
39inline void _fast_sort4(RET const in[4], RET out[4])
40{
41 RET n01 = std::min(in[1], in[0]);
42 RET x01 = std::max(in[0], in[1]);
43 RET n23 = std::min(in[2], in[3]);
44 RET x23 = std::max(in[3], in[2]);
45 RET r0 = std::min(n01, n23);
46 RET x02 = std::max(n23, n01);
47 RET n13 = std::min(x01, x23);
48 RET r3 = std::max(x23, x01);
49 RET r1 = std::min(x02, n13);
50 RET r2 = std::max(n13, x02);
51 if(ascending) { out[0] = r0; out[1] = r1; out[2] = r2; out[3] = r3; }
52 else { out[0] = r3; out[1] = r2; out[2] = r1; out[3] = r0; }
53}
