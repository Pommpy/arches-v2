120 uint64_t get_next(uint8_t& offset, uint8_t& size)
121 {
122 assert(type == _LFB::Type::WRITE_COMBINING);
123
124 //try to find largest contiguous alligned block to issue store
125 size = CACHE_BLOCK_SIZE;
126 while(size > 0)
127 {
128 uint64_t comb_mask = generate_nbit_mask(size);
129 for(offset = 0; offset < CACHE_BLOCK_SIZE; offset += size)
130 {
131 if((comb_mask & byte_mask) == comb_mask)
132 {
133 byte_mask &= ~(comb_mask);
134 return 1;
135 }
136
137 comb_mask <<= size;
138 }
139 size /= 2;
140 }
